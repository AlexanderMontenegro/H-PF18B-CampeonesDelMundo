Usuario
-------
	id (PK)
	nombre	
	correo electrónico
	contraseña
	rol (user/admin)
	fecha_creacion
	fecha_modificacion

Producto
--------
	id (PK)
	nombre
	descripción
	precio
	imagen_url
	categoria_id (FK)
	stock
	activo

Categoría
---------
	id (PK)
	nombre

Pedido
------
	id (PK)
	usuario_id (FK)
	fecha_pedido
	estado
	total

DetallePedido
-------------
	id (PK)
	pedido_id (FK)
	producto_id (FK)
	cantidad
	precio

Reseña
------
	id (PK)
	usuario_id (FK)
	producto_id (FK)
	puntuación
	comentario
	fecha_creacion

Notificación
------------
	id (PK)
	usuario_id (FK)
	mensaje
	leida
	fecha_creacion
